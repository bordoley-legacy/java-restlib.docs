001package restlib.ext.protobuf;
002
003import restlib.ContentInfo;
004import restlib.Request;
005import restlib.Response;
006import restlib.bio.OutputStreamSerializer;
007import restlib.data.MediaRange;
008import restlib.server.bio.OutputStreamSerializerSupplier;
009
010import com.google.protobuf.Message;
011
012public final class ProtobufServerSerializerSupplier {
013    public static OutputStreamSerializerSupplier create(
014            final Message prototype, final MediaRange mediaRange) {
015        final ContentInfo contentInfo = ContentInfo.builder()
016                .setMediaRange(mediaRange).build();
017
018        return new OutputStreamSerializerSupplier() {
019            @Override
020            public MediaRange mediaRange() {
021                return mediaRange;
022            }
023
024            @Override
025            public OutputStreamSerializer get(final Request request,
026                    final Response response) {
027                return ProtobufSerializer.create(
028                        prototype, response.entity().get(), contentInfo);
029            }
030        };
031    }
032
033    private ProtobufServerSerializerSupplier() {
034    }
035}
