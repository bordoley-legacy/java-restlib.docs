001package restlib.ext.jackson;
002
003import java.io.BufferedWriter;
004import java.io.IOException;
005import java.io.OutputStream;
006import java.io.OutputStreamWriter;
007import java.io.Writer;
008
009import org.codehaus.jackson.JsonGenerationException;
010import org.codehaus.jackson.map.JsonMappingException;
011import org.codehaus.jackson.map.ObjectWriter;
012
013import restlib.ContentInfo;
014import restlib.bio.OutputStreamSerializer;
015import restlib.data.Charset;
016
017import com.google.common.base.Preconditions;
018import com.google.common.io.CountingOutputStream;
019
020public final class JsonSerializer {
021    public static OutputStreamSerializer create(
022            final ObjectWriter writer,
023            final Object entity, 
024            final ContentInfo contentInfo) {
025        return new OutputStreamSerializer() {
026            @Override
027            public ContentInfo contentInfo() {
028                return contentInfo;
029            }
030
031            @Override
032            public long write(final OutputStream os) throws IOException {
033                Preconditions.checkNotNull(os);
034
035                final Charset charset = contentInfo.mediaRange().isPresent() ?
036                        contentInfo.mediaRange().get().charset().or(Charset.UTF_8) :
037                            Charset.UTF_8;
038                
039                final CountingOutputStream cos = new CountingOutputStream(os);
040                final Writer bWriter = new BufferedWriter(
041                        new OutputStreamWriter(cos, charset.toNioCharset()));
042
043                try {
044                    writer.writeValue(bWriter, entity);
045                    return cos.getCount();
046                } catch (final JsonGenerationException e) {
047                    throw new IOException(e);
048                } catch (final JsonMappingException e) {
049                    throw new IOException(e);
050                }
051            }
052        };
053    }
054
055    private JsonSerializer() {
056    }
057}
