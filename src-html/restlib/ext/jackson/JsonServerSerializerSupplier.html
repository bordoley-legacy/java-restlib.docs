001package restlib.ext.jackson;
002
003import java.util.Set;
004
005import org.codehaus.jackson.map.ObjectMapper;
006import org.codehaus.jackson.map.ObjectWriter;
007
008import restlib.ContentInfo;
009import restlib.Request;
010import restlib.Response;
011import restlib.bio.OutputStreamSerializer;
012import restlib.data.Charset;
013import restlib.data.Language;
014import restlib.data.MediaRange;
015import restlib.data.Preference;
016import restlib.server.bio.OutputStreamSerializerSupplier;
017
018import com.google.common.base.Preconditions;
019import com.google.common.collect.ImmutableSet;
020
021public final class JsonServerSerializerSupplier {
022    public static OutputStreamSerializerSupplier create(
023            final ObjectMapper objectMapper, final MediaRange mediaRange) {
024        return create(objectMapper, mediaRange, ImmutableSet.<Language> of());
025    }
026
027    public static OutputStreamSerializerSupplier create(
028            final ObjectMapper objectMapper, 
029            final MediaRange mediaRange,
030            final Iterable<Language> languages) {
031        Preconditions.checkNotNull(objectMapper);
032        final ObjectWriter writer = objectMapper.writer();
033        return create(writer, mediaRange, languages);
034    }
035
036    public static OutputStreamSerializerSupplier create(
037            final ObjectWriter writer, final MediaRange mediaRange) {
038        return create(writer, mediaRange, ImmutableSet.<Language> of());
039    }
040
041    public static OutputStreamSerializerSupplier create(
042            final ObjectWriter writer, 
043            final MediaRange mediaRange,
044            Iterable<Language> languages) {
045        Preconditions.checkNotNull(writer);
046        Preconditions.checkNotNull(mediaRange);
047        final Set<Language> langCopy = ImmutableSet.copyOf(languages);
048
049        return new OutputStreamSerializerSupplier() {
050            @Override
051            public OutputStreamSerializer get(final Request request,
052                    final Response response) {
053                final Charset charset = 
054                        Preference.<Charset> bestMatch(
055                                request.preferences().acceptedCharsets(), 
056                                Charset.available()).or(Charset.UTF_8);
057
058                final ContentInfo contentInfo = 
059                        ContentInfo.builder()
060                            .setMediaRange(mediaRange.withCharset(charset))
061                            .addLanguages(langCopy).build();
062
063                return JsonSerializer.create(
064                        writer, response.entity().get(), contentInfo);
065            }
066
067            @Override
068            public MediaRange mediaRange() {
069                return mediaRange;
070            }
071        };
072    }
073
074    private JsonServerSerializerSupplier() {
075    }
076}
