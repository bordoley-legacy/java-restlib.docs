001package restlib.ext.jackson;
002
003import java.io.IOException;
004import java.io.InputStream;
005
006import org.codehaus.jackson.JsonParseException;
007import org.codehaus.jackson.map.JsonMappingException;
008import org.codehaus.jackson.map.ObjectReader;
009
010import restlib.bio.InputStreamDeserializer;
011
012import com.google.common.base.Preconditions;
013
014public final class JsonDeserializer {
015    public static <T> InputStreamDeserializer<T> create(
016            final ObjectReader reader, final Class<T> clss) {
017        return new InputStreamDeserializer<T>() {
018            @Override
019            public T read(final InputStream is) throws IOException {
020                Preconditions.checkNotNull(is);
021                try {
022                    return clss.cast(reader.readValue(is));
023                } catch (final JsonParseException e) {
024                    throw new IOException(e);
025                } catch (final JsonMappingException e) {
026                    throw new RuntimeException(e);
027                }
028            }
029        };
030    }
031
032    private JsonDeserializer() {
033    }
034}
