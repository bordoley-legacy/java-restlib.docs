075/**
076 * Object representation of an HTTP Request. 
077 * Implementations must be immutable or effectively immutable.
078 */
079@Immutable
080public abstract class Request {        
081    /**
082     * Returns a new RequestBuilder instance.
083     */
084    public static RequestBuilder builder() {
085        return new RequestBuilder();
086    }
087    
088    /**
089     * Parses a Request from the String representation of its components.
090     * @param uriScheme the URI scheme of the request. Usually http or https.
091     * @param methodName the request method name from the request line.
092     * @param requestTarget the request target from the request line.
093     * @param headers an Iterable of the header field name/value pairs.
094     * @throws NullPointerException if any argument is null. Also if any entry 
095     * in {@code headers} is null or contains null key/values.
096     */
097    public static Request parse(
098            final String uriScheme,
099            final Method method, 
100            final Uri requestTarget, 
101            final Iterable<? extends Entry<Header, String>> headers) {      
102        Preconditions.checkNotNull(uriScheme);
103        Preconditions.checkNotNull(method);
104        Preconditions.checkNotNull(requestTarget);
105        Preconditions.checkNotNull(headers);
106        
107        final RequestBuilder requestBuilder = Request.builder();
108        final ClientPreferencesBuilder preferencesBuilder = ClientPreferences.builder();
109        final ConnectionInfoBuilder connectionInfoBuilder = ConnectionInfo.builder();
110        final ContentInfoBuilder contentInfoBuilder = ContentInfo.builder();
111        final RequestPreconditionsBuilder preconditionsBuilder = RequestPreconditions.builder();
112        
113        String host = "";
114        
115        final Function<Header, Void> checkIfHeaderIsUnique = new Function<Header,Void>() {
116            final Set<Header> uniqueHeaders = Sets.newHashSet();
117            
118            @Override
119            public Void apply(final Header fieldName) {
120                Preconditions.checkArgument(!uniqueHeaders.contains(fieldName), 
121                        "Request may not include the header " + fieldName.toString() + " more than once.");
122                uniqueHeaders.add(fieldName);         
123                return null;
124            }           
125        };    
126        
127        for (final Entry<Header, String> header : headers) {
128            Preconditions.checkNotNull(header);
129            
130            final Header fieldName = header.getKey();
131            final String fieldValue = Preconditions.checkNotNull(header.getValue());
132            
133            if (fieldName.equals(ACCEPT)) {
134                preferencesBuilder.addAcceptedMediaRanges(
135                        HttpHeaders.parseAccept(fieldValue));
136            } else if (fieldName.equals(ACCEPT_CHARSET)) {  
137                preferencesBuilder.addAcceptedCharsets(
138                        HttpHeaders.parseAcceptCharset(fieldValue));
139            } else if (fieldName.equals(ACCEPT_ENCODING)) {   
140                preferencesBuilder.addAcceptedEncodings(
141                        HttpHeaders.parseAcceptEncoding(fieldValue));
142            } else if (fieldName.equals(ACCEPT_LANGUAGE)) {  
143                preferencesBuilder.addAcceptedLanguages(
144                        HttpHeaders.parseAcceptLanguage(fieldValue));  
145            } else if (fieldName.equals(AUTHORIZATION)) {
146                checkIfHeaderIsUnique.apply(fieldName);                
147                requestBuilder.setAuthorizationCredentials(
148                        HttpHeaders.parsetAuthorization(fieldValue));
149            } else if (fieldName.equals(CACHE_CONTROL)) {   
150                requestBuilder.addCacheDirectives(
151                        HttpHeaders.parseCacheControl(fieldValue));
152            } else if (fieldName.equals(CONNECTION)) {  
153                connectionInfoBuilder.addConnectionOptions(
154                        HttpHeaders.parseConnection(fieldValue));
155            } else if (fieldName.equals(CONTENT_ENCODING)) {
156                contentInfoBuilder.addEncodings(
157                        HttpHeaders.parseContentEncoding(fieldValue));
158            } else if (fieldName.equals(CONTENT_LANGUAGE)) {
159                contentInfoBuilder.addLanguages(
160                        HttpHeaders.parseContentLanguage(fieldValue));
161            } else if (fieldName.equals(CONTENT_LENGTH)) { 
162                checkIfHeaderIsUnique.apply(fieldName);                
163                contentInfoBuilder.setLength(
164                            HttpHeaders.parseContentLength(fieldValue));
165            } else if (fieldName.equals(CONTENT_LOCATION)) { 
166                checkIfHeaderIsUnique.apply(fieldName);               
167                contentInfoBuilder.setLocation(
168                        HttpHeaders.parseContentLocation(fieldValue));
169            } else if (fieldName.equals(CONTENT_TYPE)) { 
170                checkIfHeaderIsUnique.apply(fieldName);                
171                contentInfoBuilder.setMediaRange(
172                        HttpHeaders.parseContentType(fieldValue));
173            } else if (fieldName.equals(EXPECT)) {
174                requestBuilder.addExpectations(
175                        HttpHeaders.parseExpect(fieldValue));
176            } else if (fieldName.equals(FROM)) {  
177                checkIfHeaderIsUnique.apply(fieldName);               
178                requestBuilder.setFrom(HttpHeaders.parseFrom(fieldValue));
179            } else if (fieldName.equals(HOST)) {
180                checkIfHeaderIsUnique.apply(fieldName);             
181                host = fieldValue;
182            } else if (fieldName.equals(IF_MATCH)) {    
183                preconditionsBuilder.addIfMatchTags(
184                        HttpHeaders.parseIfMatch(fieldValue));
185            } else if (fieldName.equals(IF_MODIFIED_SINCE)) {  
186                checkIfHeaderIsUnique.apply(fieldName);              
187                preconditionsBuilder.setIfModifiedSinceDate(
188                        HttpHeaders.parseIfModifiedSince(fieldValue));
189            } else if (fieldName.equals(IF_NONE_MATCH)) { 
190                preconditionsBuilder.addIfNoneMatchTags(
191                        HttpHeaders.parseIfNoneMatch(fieldValue));
192            } else if (fieldName.equals(IF_RANGE)) {
193                checkIfHeaderIsUnique.apply(fieldName);                
194                final Object ifRange = 
195                        HttpHeaders.parseIfRange(fieldValue);
196                preconditionsBuilder.setIfRange(ifRange);
197            } else if (fieldName.equals(IF_UNMODIFIED_SINCE)) {  
198                checkIfHeaderIsUnique.apply(fieldName);               
199                preconditionsBuilder.setIfUnmodifiedSinceDate(
200                        HttpHeaders.parseIfUnmodifiedSince(fieldValue));
201            } else if (fieldName.equals(MAX_FORWARDS)) { 
202                checkIfHeaderIsUnique.apply(fieldName);              
203                requestBuilder.setMaxForwards(
204                        HttpHeaders.parseMaxForwards(fieldValue));
205            } else if (fieldName.equals(PRAGMA)) {  
206                requestBuilder.addPragmaCacheDirectives(
207                        HttpHeaders.parsePragma(fieldValue));
208            } else if (fieldName.equals(PROXY_AUTHORIZATION)) { 
209                checkIfHeaderIsUnique.apply(fieldName);       
210                requestBuilder.setProxyAuthorizationCredentials(
211                        HttpHeaders.parseProxyAuthorization(fieldValue));
212            } else if (fieldName.equals(RANGE)) {
213                checkIfHeaderIsUnique.apply(fieldName);               
214                preferencesBuilder.setRange(
215                        HttpHeaders.parseRange(fieldValue));
216            } else if (fieldName.equals(REFERER)) {  
217                checkIfHeaderIsUnique.apply(fieldName);         
218                requestBuilder.setReferrer(
219                        HttpHeaders.parseReferer(fieldValue));
220            } else if (fieldName.equals(TE)) { 
221                preferencesBuilder.addAcceptedTransferEncodings(
222                        HttpHeaders.parseTE(fieldValue));
223            } else if (fieldName.equals(TRAILER)) {  
224                connectionInfoBuilder.addTrailerHeaders(
225                        HttpHeaders.parseTrailer(fieldValue));
226            } else if (fieldName.equals(TRANSFER_ENCODING)) { 
227                connectionInfoBuilder.addTransferEncodings(
228                        HttpHeaders.parseTransferEncoding(fieldValue));
229            } else if (fieldName.equals(UPGRADE)) {
230                connectionInfoBuilder.addUpgradeProtocols(
231                        HttpHeaders.parseUpgrade(fieldValue));
232            } else if (fieldName.equals(USER_AGENT)) {  
233                checkIfHeaderIsUnique.apply(fieldName);          
234                requestBuilder.setUserAgent(
235                        HttpHeaders.parseUserAgent(fieldValue));
236            } else if (fieldName.equals(VIA)) { 
237                connectionInfoBuilder.addVias(
238                        HttpHeaders.parseVia(fieldValue));
239            } else {
240                requestBuilder.addCustomHeader(fieldName, fieldValue);
241            }    
242        }
243        
244        final Uri uri;
245        if (!requestTarget.isAbsolute()) {
246            Preconditions.checkArgument(
247                    !host.isEmpty(), "Uri is not absolute, and does not include a host field.");
248            uri = Uri.builder()
249                    .setScheme(uriScheme)
250                    .setAuthority(host)
251                    .setPath(requestTarget.path())
252                    .setQuery(requestTarget.query())
253                    .build();
254        } else {
255                Preconditions.checkArgument(host.equals(""), "Host header present when request target is an absolute URI.");
256            uri = requestTarget;
257        }
258        
259        return requestBuilder
260                .setMethod(method)
261                .setUri(uri)
262                .setPreferences(preferencesBuilder.build())
263                .setConnectionInfo(connectionInfoBuilder.build())
264                .setContentInfo(contentInfoBuilder.build())
265                .setPreconditions(preconditionsBuilder.build()).build();
266    }
267    
268    /**
269     * Parses a Request from the String representation of its components.
270     * @param uriScheme the URI scheme of the request. Usually http or https.
271     * @param methodName the request method name from the request line.
272     * @param requestTarget the request target from the request line.
273     * @param headers an Iterable of the header field name/value pairs.
274     * @throws NullPointerException if any argument is null. Also if any entry 
275     * in {@code headers} is null or contains null key/values.
276     */
277    public static Request parse(
278            final String uriScheme,
279            final String method, 
280            final String requestTarget, 
281            final Iterable<? extends Entry<String, String>> headers) {  
282        Preconditions.checkNotNull(uriScheme);
283        Preconditions.checkNotNull(method);
284        Preconditions.checkNotNull(requestTarget);
285        Preconditions.checkNotNull(headers);
286        return parse(uriScheme, Method.forName(method), Uri.parse(requestTarget), 
287                        Iterables.transform(headers,
288                                        new Function<Entry<String, String>, Entry<Header, String>>(){
289                                                        @Override
290                                                        public Entry<Header, String> apply(
291                                                                        final Entry<String, String> input) {
292                                                                Preconditions.checkNotNull(input);
293                                                                return ImmutableMapEntry.create(
294                                                                                Header.create(input.getKey()), input.getValue());
295                                                        }                               
296                        }));
297    }

377    @Override
378    public final String toString() {
379        final StringBuilder builder = new StringBuilder();
380
381        builder.append(this.method()).append(" ").append(this.uri().path());
382        if (!this.uri().query().isEmpty()) {
383            builder.append('?').append(this.uri().query());
384        }
385        builder.append("\r\n");
386
387        appendHeader(builder, HttpHeaders.HOST, this.uri().authority());
388        appendHeader(builder, HttpHeaders.AUTHORIZATION, this.authorizationCredentials());
389        appendHeader(builder, HttpHeaders.CACHE_CONTROL, this.cacheDirectives());
390        appendHeader(builder, HttpHeaders.EXPECT, this.expectations());
391        appendHeader(builder, HttpHeaders.FROM, this.from());
392        appendHeader(builder, HttpHeaders.MAX_FORWARDS, this.maxForwards());
393        appendHeader(builder, HttpHeaders.PRAGMA, this.pragmaCacheDirectives());
394        appendHeader(builder, HttpHeaders.PROXY_AUTHORIZATION, this.proxyAuthorizationCredentials());
395        appendHeader(builder, HttpHeaders.REFERER, this.referrer());
396        appendHeader(builder, HttpHeaders.USER_AGENT, this.userAgent());
397
398        return builder
399                .append(this.connectionInfo())
400                .append(this.contentInfo())
401                .append(this.preconditions())
402                .append(this.preferences())
403                .toString();
404    }
