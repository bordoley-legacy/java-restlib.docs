018package restlib.serializable.atom;
019
020import restlib.data.Language;
021import restlib.data.MediaRange;
022import restlib.net.IRI;
023
024import com.google.common.base.Optional;
025import com.google.common.base.Preconditions;
026
027public final class AtomLinkBuilder {    
028    IRI href = null;
029    Optional<String> rel = Optional.absent();
030    Optional<MediaRange> type = Optional.absent();
031    Optional<Language> hrefLang = Optional.absent();
032    Optional<String> title = Optional.absent();
033    Optional<Long> length = Optional.absent();
034    
035    AtomLinkBuilder() {
036    }
037    
038    public AtomLink build() {
039        Preconditions.checkState(this.href != null);
040        return new AtomLinkImpl(this);
041    }
042    
043    public AtomLinkBuilder setHref(final IRI href) {
044        Preconditions.checkNotNull(href);
045        this.href = href;
046        return this;
047    }
048
049    public AtomLinkBuilder setRel(final String rel) {
050        this.rel = Optional.of(rel);
051        return this;
052    }
053
054    public AtomLinkBuilder setType(final MediaRange type) {
055        this.type = Optional.of(type);
056        return this;
057    }
058    
059    public AtomLinkBuilder setHrefLang(final Language hrefLang) {
060        this.hrefLang = Optional.of(hrefLang);
061        return this;
062    }
063
064    public AtomLinkBuilder setTitle(final String title) {
065        this.title  = Optional.of(title);
066        return this;
067    }
068    
069    public AtomLinkBuilder setLength(final long length) {
070        Preconditions.checkArgument(length >= 0);
071        this.length = Optional.of(length);
072        return this;
073    }
074}
