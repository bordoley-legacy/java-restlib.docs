001package restlib.server.bio;
002
003import com.google.common.base.Preconditions;
004
005public final class BioResources {
006    public static final BioResource<?> NOT_FOUND = new BioNotFoundResource();
007    
008    public static <T> BioResource<T> contentEncodingResource(final BioResource<T> resource, final int bufSize) {
009        Preconditions.checkNotNull(resource);
010        Preconditions.checkArgument(bufSize > 0);
011        
012        return new BioContentEncodingResource<T>(resource, bufSize);     
013    }
014    
015    public static <T> BioResource<T> errorResource(final BioResource<T> resource) {
016        Preconditions.checkNotNull(resource);
017        return new BioErrorResource<T>(resource);
018    }
019    
020    public static <T> BioResource<T> limitInputResource(final BioResource<T> resource, final long limit) {
021        Preconditions.checkNotNull(resource);
022        Preconditions.checkArgument(limit >= 0);
023        
024        return new LimitInputResource<T>(resource, limit); 
025    }
026    
027    private BioResources(){}
028}
