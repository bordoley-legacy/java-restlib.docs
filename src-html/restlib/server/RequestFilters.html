001package restlib.server;
002
003import restlib.Request;
004import restlib.data.ExtensionMap;
005import restlib.data.Header;
006
007import com.google.common.base.Function;
008import com.google.common.base.Preconditions;
009import com.google.common.collect.ImmutableSet;
010
011public final class RequestFilters {
012    public static Function<Request, Request> DEFAULT_EXTENSION_FILTER =
013            ExtensionRequestFilter.getDefaultInstance();
014            
015    public static Function<Request, Request> METHOD_OVERRIDE = new MethodRequestFilter();
016    
017    public static Function<Request, Request> queryFilter(final Iterable<Header> headers) {  
018        return new QueryRequestFilter(ImmutableSet.copyOf(headers));   
019    };
020    
021    public static Function<Request, Request> extensionFilter(final ExtensionMap extensionMap) {
022        Preconditions.checkNotNull(extensionMap);
023        return new ExtensionRequestFilter(extensionMap);
024    }
025 
026    private RequestFilters(){}
027}
