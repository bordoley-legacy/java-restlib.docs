001package restlib.impl;
002
003import java.util.Map.Entry;
004
005import javax.annotation.Nullable;
006import javax.annotation.concurrent.Immutable;
007
008import com.google.common.base.Objects;
009import com.google.common.base.Preconditions;
010
011/**
012 * A {@code Map.Entry} that maintains an immutable key and value.
013 */
014@Immutable
015public final class ImmutableMapEntry<K, V> implements Entry<K,V> {
016    /**
017     * Returns a new {@code ImmutablMapEntry} representing the mapping of key to value.
018     * @param key a non-null key
019     * @param value a non-null value
020     * @throws NullPointerException if key is null.
021     */
022    public static <K,V> ImmutableMapEntry<K,V> create(final K key, final V value) {
023        Preconditions.checkNotNull(key);
024        Preconditions.checkNotNull(value);
025        return new ImmutableMapEntry<K,V>(key,value);
026    }
027    
028    private final K key;
029    private final V value;
030    
031    private ImmutableMapEntry(final K key, final V value) {
032        this.key = key;
033        this.value = value;
034    }
035
036    @Override
037    public boolean equals(@Nullable final Object obj) {
038        if (this == obj) {
039            return true;
040        } else if (obj instanceof Entry) {
041            final Entry<?, ?> that = (Entry<?, ?>) obj;
042            return Objects.equal(this.getKey(), that.getKey())
043                    && Objects.equal(this.getValue(), that.getValue());
044        } else {
045            return false;
046        }
047    }
048    @Override
049    public K getKey() {
050        return key;
051    }
052
053    @Override
054    public V getValue() {
055        return value;
056    }
057    
058    @Override
059    public int hashCode() {
060        // Only contains non-null key and value 
061        return getKey().hashCode() ^ getValue().hashCode();
062    }
063
064    /**
065     * Guaranteed to throw an {@code UnsupportedOperationException}.
066     */
067    @Override
068    public V setValue(@Nullable final V value) {
069        throw new UnsupportedOperationException();
070    }
071
072    @Override
073    public String toString() {
074        return key + "=" + "[" + value + "]";
075    }
076}
