001package restlib.impl;
002
003import com.google.common.base.CharMatcher;
004import com.google.common.base.Preconditions;
005import com.google.common.primitives.UnsignedInts;
006import com.google.common.primitives.UnsignedLongs;
007
008/**
009 * Static utilities for parsing unsigned numbers.
010 */
011public final class CommonParsers {
012    private static final CharMatcher DIGITS = CharMatcher.inRange('0', '9');
013    
014    /**
015     * Parses a {@code String} to an unsigned {@code int}.
016     * @param in a non-null string containing only valid ASCII digits.
017     * @throws NullPointerException if {@code in} is null.
018     * @throws IllegalArgumentException if {@code in} contains any characters other than ASCII digits.
019     */
020    public static int parseUnsignedInteger(final String in) {
021        Preconditions.checkNotNull(in);
022        Preconditions.checkArgument(!in.isEmpty());
023        
024        // Java 1.7 allows Strings starting with +/- 
025        // as the leading char, we don't.
026        Preconditions.checkArgument(DIGITS.apply(in.charAt(0)));
027        return UnsignedInts.parseUnsignedInt(in);
028    }
029    
030    /**
031     * Parses a {@code String} to an unsigned {@code long}.
032     * @param in a non-null string containing only valid ASCII digits.
033     * @throws NullPointerException if {@code in} is null.
034     * @throws IllegalArgumentException if {@code in} contains any characters other than ASCII digits.
035     */
036    public static long parseUnsignedLong(final String in) {
037        Preconditions.checkNotNull(in);
038        Preconditions.checkArgument(!in.isEmpty());
039        
040        // Java 1.7 allows Strings starting with +/-
041        // as the leading char, we don't.
042        Preconditions.checkArgument(DIGITS.apply(in.charAt(0)));
043        return UnsignedLongs.parseUnsignedLong(in);
044    }
045    
046    private CommonParsers(){}
047}
