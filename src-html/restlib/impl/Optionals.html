001package restlib.impl;
002
003import com.google.common.base.Optional;
004import com.google.common.base.Preconditions;
005
006/**
007 * Static utilities methods pertaining to instances of {@code Optional}. 
008 */
009public final class Optionals {
010    /**
011     * Returns whether the Optional value is absent.
012     * @param optional a non-null Optional.
013     * @throws NullPointerException if {@code optional} is null.
014     */
015    public static boolean isAbsent(final Optional<?> optional) {
016        Preconditions.checkNotNull(optional);
017        return !optional.isPresent();
018    }
019    
020    /**
021     * Returns the result of call toString() on the optional value if present.
022     * @param optional a non-null Optional value that is present.
023     * @throws NullPointerException if {@code optional} is null.
024     * @throws IllegalArgumentException if optional is not present.
025     */
026    public static String toString(final Optional<?> optional) {
027        Preconditions.checkNotNull(optional);
028        Preconditions.checkArgument(optional.isPresent());
029        return optional.get().toString();
030    }
031    
032    /**
033     * Returns the result of calling toString on the optional value if present,
034     * otherwise the empty string is returned.
035     * @param optional a non-null Optional value.
036     * @throws NullPointerException if {@code optional} is null.
037     */
038    public static String toStringOrEmpty(final Optional<?> optional) {
039        Preconditions.checkNotNull(optional);
040        return optional.isPresent() ? optional.get().toString() : "";
041    }
042    
043    private Optionals(){}
044}
