015/**
016 * Status of an HTTP Response.
017 */
018@Immutable
019public final class Status {
020        /**
021         * Enumeration representing the different classes of HTTP response statuses.
022         */
023    public static enum Class {
024        /**
025         * The request contains bad syntax or cannot be fulfilled.
026         */
027        CLIENT_ERROR,
028        
029        /**
030         * Informational - Request received, continuing process.
031         */
032        INFORMATIONAL,
033        
034        /**
035         * Further action must be taken in order to complete the request.
036         */
037        REDIRECTION,
038        
039        /**
040         * The server failed to fulfill an apparently valid request.
041         */
042        SERVER_ERROR,
043        
044        /**
045         * The action was successfully received, understood, and accepted.
046         */
047        SUCCESS,
048        
049        /**
050         * The status does not belong to a defined status class.
051         */
052        UNDEFINED;
053        
054        
055        private static Class getStatusClassForCode(final int code) {
056            if ((code >= 100) && (code < 200)) {
057                return Class.INFORMATIONAL;
058            } else if ((code >= 200) && (code < 300)) {
059                return Class.SUCCESS;
060            } else if ((code >= 300) && (code < 400)) {
061                return Class.REDIRECTION;
062            } else if ((code >= 400) && (code < 500)) {
063                return Class.CLIENT_ERROR;
064            } else if ((code >= 500) && (code < 600)) {
065                return Class.SERVER_ERROR;
066            } else {
067                return UNDEFINED;
068            }
069        }
070    }

662    /**
663     * Returns a Status instance for {@code code}
664     * @param code any valid status code between 100 and 999 inclusive.
665     * @throws IllegalArgumentException if {@code code} is not a valid status code.
666     */
667    public static Status forCode(final int code) {
668        Preconditions.checkArgument((code >= 100) && (code < 1000));
669        
670        return Objects.firstNonNull(
671                REGISTERED_STATUSES.get(code), 
672                new Status(code, "", ""));
673    }
674            
675    private static Status register(
676            final int code,
677            final String reason,
678            final String message) {
679        Preconditions.checkArgument((code >= 100) && (code < 1000));
680        Preconditions.checkNotNull(reason);
681        Preconditions.checkArgument(CharMatchers.REASON_PHRASE.matchesAllOf(reason));
682        Preconditions.checkNotNull(message);
683
684        final Status status = new Status(code, reason, message);
685        return Objects.firstNonNull(
686                REGISTERED_STATUSES.putIfAbsent(code, status), status);
687    }

690    private final String message;
691    private final String reason;
692    private final Class statusClass;

701        this.message = message;
702    }
