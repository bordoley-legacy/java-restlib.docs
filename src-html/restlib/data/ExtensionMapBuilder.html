001package restlib.data;
002
003import com.google.common.base.Preconditions;
004import com.google.common.collect.ImmutableBiMap;
005
006public final class ExtensionMapBuilder {
007    private final ImmutableBiMap.Builder<MediaRange, String> mediaRangeExtMap = ImmutableBiMap.builder();
008    
009    ExtensionMapBuilder(){
010    }
011    
012    public ExtensionMap build() {
013        return new ExtensionMap(this.mediaRangeExtMap.build());
014    }
015    
016    public ExtensionMapBuilder put(final MediaRange mediaRange, final String ext) {
017        Preconditions.checkNotNull(mediaRange);
018        Preconditions.checkNotNull(ext);
019        Preconditions.checkArgument(!ext.isEmpty());
020        
021        this.mediaRangeExtMap.put(mediaRange, ext);
022        return this;
023    }
024}
